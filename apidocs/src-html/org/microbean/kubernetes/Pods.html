039/**
040 * A utility class that helps with operations concerning {@link Pod}s.
041 *
042 * @author <a href="https://about.me/lairdnelson"
043 * target="_parent">Laird Nelson</a>
044 *
045 * @see #forwardPort(OkHttpClient, Listable, int)
046 *
047 * @see #isReady(Pod)
048 *
049 * @see Pod
050 */
051public final class Pods {
052
053  /**
054   * Forwards an arbitrary local port to the supplied {@code
055   * remotePort} on the {@linkplain #getFirstReadyPod(Listable) first
056   * ready <code>Pod</code>} in the supplied {@link Listable} using
057   * the supplied {@link OkHttpClient}.
058   *
059   * <p>This method may return {@code null}.</p>
060   *
061   * @param httpClient an {@link OkHttpClient} used to communicate
062   * with Kubernetes; may be {@code null} in which case {@code null}
063   * will be returned
064   *
065   * @param pods a {@link Listable} of {@link PodList}s; <strong>must
066   * also be an instance of {@link OperationSupport}</strong> or
067   * {@code null} will be returned
068   *
069   * @param remotePort the port on the {@link Pod} that is ultimately
070   * located to which traffic should be forwarded
071   *
072   * @return a {@link LocalPortForward} representing the port
073   * forwarding operation, or {@code null}; must be {@linkplain
074   * Closeable#close() closed} if non-{@code null}
075   *
076   * @exception MalformedURLException if a {@link URL} to the
077   * Kubernetes resource representing the {@link Pod} in question
078   * could not be constructed
079   *
080   * @see PortForwarderWebsocket#PortForwarderWebsocket(OkHttpClient)
081   *
082   * @see <a
083   * href="https://kubernetes.io/docs/concepts/workloads/pods/pod-lifecycle/#pod-conditions">Pod
084   * conditions documentation</a>
085   */
086  public static final LocalPortForward forwardPort(final OkHttpClient httpClient, final Listable<? extends PodList> pods, final int remotePort) throws MalformedURLException {
087    LocalPortForward returnValue = null;
088    if (httpClient != null && pods instanceof OperationSupport) {
089      final String urlBase = ((OperationSupport)pods).getNamespacedUrl().toExternalForm();
090      assert urlBase != null;
091      final Pod readyPod = getFirstReadyPod(pods);
092      if (readyPod != null) {
093        final String name = readyPod.getMetadata().getName();
094        assert name != null;
095        final URL url = new URL(new StringBuilder(urlBase).append("/").append(name).toString());
096        returnValue = new PortForwarderWebsocket(httpClient).forward(url, remotePort);
097      }
098    }
099    return returnValue;
100  }

102  /**
103   * Returns {@link Boolean#TRUE} if the supplied {@link Pod} is known
104   * to be ready, {@link Boolean#FALSE} if it is known to be not ready
105   * and {@code null} if its readiness status is unknown.
106   *
107   * <p>This method may return {@code null}.</p>
108   *
109   * @param pod the {@link Pod} to check; may be {@code null} in which
110   * case {@code null} will be returned
111   *
112   * @return {@link Boolean#TRUE} if the supplied {@link Pod} is known
113   * to be ready, {@link Boolean#FALSE} if it is known to be not ready
114   * and {@code null} if its readiness status is unknown
115   *
116   * @see #isReady(PodStatus)
117   *
118   * @see <a
119   * href="https://kubernetes.io/docs/concepts/workloads/pods/pod-lifecycle/#pod-conditions">Pod
120   * conditions documentation</a>
121   */
122  public static final Boolean isReady(final Pod pod) {
123    Boolean returnValue = null;
124    if (pod != null) {
125      returnValue = isReady(pod.getStatus());
126    }
127    return returnValue;
128  }
129
130  /**
131   * Returns {@link Boolean#TRUE} if the supplied {@link PodStatus} is
132   * known to be ready, {@link Boolean#FALSE} if it is known to be not
133   * ready and {@code null} if its readiness status is unknown.
134   *
135   * <p>This method may return {@code null}.</p>
136   *
137   * @param podStatus the {@link PodStatus} to check; may be {@code
138   * null} in which case {@code null} will be returned
139   *
140   * @return {@link Boolean#TRUE} if the supplied {@link PodStatus} is
141   * known to be ready, {@link Boolean#FALSE} if it is known to be not
142   * ready and {@code null} if its readiness status is unknown
143   *
144   * @see #isReady(Iterable)
145   *
146   * @see <a
147   * href="https://kubernetes.io/docs/concepts/workloads/pods/pod-lifecycle/#pod-conditions">Pod
148   * conditions documentation</a>
149   */
150  public static final Boolean isReady(final PodStatus podStatus) {
151    Boolean returnValue = null;
152    if (podStatus != null) {
153      returnValue = isReady(podStatus.getConditions());
154    }
155    return returnValue;
156  }
157
158  /**
159   * Returns {@link Boolean#TRUE} if the supplied {@link Iterable} of
160   * {@link PodCondition}s contains a {@link PodCondition} whose
161   * {@link PodCondition#getType()} method returns {@code Ready} and
162   * whose {@link PodCondition#getStatus()} method returns {@code
163   * True}.
164   *
165   * <p>If the supplied {@link Iterable} of
166   * {@link PodCondition}s <em>also</em> contains a {@link PodCondition} whose
167   * {@link PodCondition#getType()} method returns {@code Ready} and
168   * whose {@link PodCondition#getStatus()} method returns {@code
169   * False}, then {@code null} is returned.</p>
170   *
171   * <p>If instead the supplied {@link Iterable} of
172   * {@link PodCondition}s contains a {@link PodCondition} whose
173   * {@link PodCondition#getType()} method returns {@code Ready} and
174   * whose {@link PodCondition#getStatus()} method returns {@code
175   * False} and does not <em>also</em> contains a {@link PodCondition} whose
176   * {@link PodCondition#getType()} method returns {@code Ready} and
177   * whose {@link PodCondition#getStatus()} method returns {@code
178   * True}, then {@link Boolean#FALSE} is returned.</p>
179   *
180   * <p>{@code null} is returned in all other cases.</p>
181   *
182   * <p>This method may return {@code null}.</p>
183   *
184   * @param podConditions an {@link Iterable} of {@link
185   * PodCondition}s; may be {@code null} in which case {@code null}
186   * will be returned
187   *
188   * @return {@link Boolean#TRUE} if the supplied {@link Iterable} of
189   * {@link PodCondition}s represents a state of affairs known to
190   * represent the readiness of the {@link PodStatus} they describe;
191   * {@link Boolean#FALSE} if the supplied {@link Iterable} of {@link
192   * PodCondition}s represents a state of affairs known to represent
193   * the unreadiness of the {@link PodStatus} they describe, or {@code
194   * null} in all other cases
195   *
196   * @see <a
197   * href="https://kubernetes.io/docs/concepts/workloads/pods/pod-lifecycle/#pod-conditions">Pod
198   * conditions documentation</a>
199   */
200  public static final Boolean isReady(final Iterable<? extends PodCondition> podConditions) {
201    Boolean returnValue = null;
202    if (podConditions != null) {
203      for (final PodCondition condition : podConditions) {
204        if (condition != null) {
205          final String conditionType = condition.getType();
206          // See https://kubernetes.io/docs/concepts/workloads/pods/pod-lifecycle/#pod-conditions.
207          if ("Ready".equals(conditionType)) {
208            final String conditionStatus = condition.getStatus();
209            switch (conditionStatus) {
210            case "True":
211              if (Boolean.FALSE.equals(returnValue)) {
212                returnValue = null;
213              } else {
214                returnValue = Boolean.TRUE;
215              }
216              break;
217            case "False":
218              if (Boolean.TRUE.equals(returnValue)) {
219                returnValue = null;
220              } else {
221                returnValue = Boolean.FALSE;
222              }
223              break;
224            case "Unknown":
225              returnValue = null;
226              break;
227            default:
228              throw new IllegalStateException("Unexpected value for PodCondition of type Ready: " + conditionStatus);
229            }
230          }
231        }
232      }
233    }
234    return returnValue;
235  }

237  /**
238   * Returns the first {@link Pod} encountered in the supplied {@link
239   * Iterable} of {@link Pod}s for which the {@link #isReady(Pod)}
240   * method returns {@link Boolean#TRUE}, or {@code null}.
241   *
242   * <p>This method may return {@code null}.</p>
243   *
244   * @param pods an {@link Iterable} of {@link Pod}s to check; may be
245   * {@code null} in which case {@code null} will be returned
246   *
247   * @return a {@link Pod} known to be ready, or {@code null}
248   *
249   * @see #isReady(Pod)
250   *
251   * @see <a
252   * href="https://kubernetes.io/docs/concepts/workloads/pods/pod-lifecycle/#pod-conditions">Pod
253   * conditions documentation</a>
254   */
255  public static final Pod getFirstReadyPod(final Iterable<? extends Pod> pods) {
256    Pod returnValue = null;
257    if (pods != null) {
258      for (final Pod pod : pods) {
259        if (Boolean.TRUE.equals(isReady(pod))) {
260          returnValue = pod;
261          break;
262        }
263      }
264    }
265    return returnValue;
266  }
267
268  /**
269   * Returns the first {@link Pod} encountered in the supplied {@link
270   * PodList} of {@link Pod}s for which the {@link #isReady(Pod)}
271   * method returns {@link Boolean#TRUE}, or {@code null}.
272   *
273   * <p>This method may return {@code null}.</p>
274   *
275   * @param podList a {@link PodList} of {@link Pod}s to check; may be
276   * {@code null} in which case {@code null} will be returned
277   *
278   * @return a {@link Pod} known to be ready, or {@code null}
279   *
280   * @see #getFirstReadyPod(Iterable)
281   *
282   * @see #isReady(Pod)
283   *
284   * @see <a
285   * href="https://kubernetes.io/docs/concepts/workloads/pods/pod-lifecycle/#pod-conditions">Pod
286   * conditions documentation</a>
287   */
288  public static final Pod getFirstReadyPod(final PodList podList) {
289    Pod returnValue = null;
290    if (podList != null) {
291      returnValue = getFirstReadyPod(podList.getItems());
292    }
293    return returnValue;
294  }

296  /**
297   * Returns the first {@link Pod} reachable from the supplied {@link
298   * Listable} of {@link PodList}s for which the {@link #isReady(Pod)}
299   * method returns {@link Boolean#TRUE}, or {@code null}.
300   *
301   * <p>This method may return {@code null}.</p>
302   *
303   * @param podsResource a {@link Listable} of {@link PodList}s
304   * representing {@link Pod}s to check; may be {@code null} in which
305   * case {@code null} will be returned
306   *
307   * @return a {@link Pod} known to be ready, or {@code null}
308   *
309   * @see #getFirstReadyPod(PodList)
310   *
311   * @see #isReady(Pod)
312   *
313   * @see <a
314   * href="https://kubernetes.io/docs/concepts/workloads/pods/pod-lifecycle/#pod-conditions">Pod
315   * conditions documentation</a>
316   */
317  public static final Pod getFirstReadyPod(final Listable<? extends PodList> podsResource) {
318    Pod returnValue = null;
319    if (podsResource != null) {
320      returnValue = getFirstReadyPod(podsResource.list());
321    }
322    return returnValue;
323  }
